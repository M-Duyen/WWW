USE employeedb

CREATE TABLE departments(
	id INT AUTO_INCREMENT PRIMARY KEY,
	NAME NVARCHAR(255)
)

CREATE TABLE employees(
	id INT AUTO_INCREMENT PRIMARY KEY,
	NAME NVARCHAR(255),
	salary DOUBLE,
	department_id INT,
	FOREIGN KEY (department_id) REFERENCES departments(id)

)
